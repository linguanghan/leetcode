```java
            cur = nxt;
```

#### [无重复字符的最长子串](https://leetcode-cn.com/problems/longest-substring-without-repeating-characters/)

```
输入: s = "abcabcbb"
输出: 3 
解释: 因为无重复字符的最长子串是 "abc"，所以其长度为 3。
```

```java
class Solution {
    public int lengthOfLongestSubstring(String s) {
        Map<Character, Integer> window = new HashMap<>();
        int left = 0;
        int right = 0;
        int max = 0;
        while(right < s.length()){
            Character c1 = s.charAt(right);
            window.put(c1, window.getOrDefault(c1, 0) + 1);
            right++;
            while(window.get(c1) > 1){
                Character c2 = s.charAt(left);
                left++;
                window.put(c2, window.get(c2)-1);
            }

            max = Math.max(max, right - left);
        }

        return max;
    }
}
```

```java
class Solution {
    public int findKthLargest(int[] nums, int k) {
        if(nums==null || nums.length < k){
            return -1;
        }

        quickSort(nums, 0, nums.length-1);

        return nums[nums.length - k];
    }

    public void quickSort(int[] nums, int left, int right){
        if(left > right){
            return;
        }

        int key = nums[left];
        int leftIndex = left;
        int rightIndex = right;

        while(left < right){

            while(left < right && nums[right] >= key){
                right--;
            }

            nums[left] = nums[right];

            while(left < right && nums[left] <= key){
                left++;
            }

            nums[right] = nums[left];
        }

        nums[left] = key;
        quickSort(nums, leftIndex, left - 1);
        quickSort(nums, right + 1, rightIndex);
    }
}
```

```java
class Solution {
    public int findKthLargest(int[] nums, int k) {
        if(nums==null || nums.length < k){
            return -1;
        }

        int heapSize = nums.length;
        buildMaxHeap(nums, heapSize);
        for(int i = nums.length - 1;i >= nums.length - k +1;i--){
            swap(nums, i, 0);
            --heapSize;
            maxHeapify(nums, 0, heapSize);
        }

        return nums[0];
    }

    public void buildMaxHeap(int[] nums, int heapSize){
        for(int i = heapSize/2;i>=0;i--){
            maxHeapify(nums, i, heapSize);
        }
    }

    public void maxHeapify(int[] nums, int i, int heapSize){
        int l = 2 * i + 1, r = 2 * i + 2, largest = i;
        if(l < heapSize && nums[l] > nums[largest]){
            largest = l;
        }

        if(r < heapSize && nums[r] > nums[largest]){
            largest = r;
        }

        if(largest!=i){
            swap(nums, largest, i);
            maxHeapify(nums, largest, heapSize);
        }
    }

    public void swap(int[] nums, int i, int j){
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
```

```java
class LRUCache {

    DoubleLinkedList cache = new DoubleLinkedList();
    Map<Integer, Node> map = new HashMap<>();
    int capacity;
    public LRUCache(int capacity) {
        this.capacity = capacity;
    }
    
    public int get(int key) {
        if(!map.containsKey(key)){
            return -1;
        }

        Node temp = map.get(key);
        cache.remove(temp);
        cache.addLast(temp);
        map.put(key, temp);
        return temp.value;
    }
    
    public void put(int key, int value) {
        Node p = new Node(key, value);
        //如果存在
        if(map.containsKey(key)){
            Node temp = map.get(key);
            cache.remove(temp);
            cache.addLast(p);
            map.put(key, p);

        }else{
            if(cache.size < capacity){
                cache.addLast(p);
                map.put(key, p);
            }else{
                Node node =  cache.removeFirst();
                map.remove(node.key);
                cache.addLast(p);
                map.put(key, p);
            }
        }
    }
}

class DoubleLinkedList{
    Node head;
    Node tail;
    int size;
    DoubleLinkedList(){
        head = new Node(0, 0);
        tail = new Node(0, 0);
        size = 0;

        head.next = tail;
        tail.pre = head; 
    }

    public void addLast(Node node){
        node.pre = tail.pre;
        node.pre.next = node;
        tail.pre = node;
        node.next = tail;
        size++;
    }

    public Node removeFirst(){
        if(size == 0){
            return null;
        }
        Node p = head.next;
        head.next = head.next.next;
        head.next.pre = head;
        size--;

        return p;
    }

    public void remove(Node node){
        node.pre.next = node.next;
        node.next.pre = node.pre;
        node.pre = null;
        node.next = null;
        size--;
    }
}
```

### 5、三数之和

```java
class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        Arrays.sort(nums);
        List<List<Integer>> res = new ArrayList<>();

        for(int i = 0;i < nums.length - 2;i++){
            int first = nums[i];
            int left = i+1;
            int right = nums.length - 1;
            if(i > 0 && nums[i] == nums[i-1]){
                continue;
            }
            int target = -nums[i];
            while(left < right){
                if(nums[left] + nums[right] == target){
                    List<Integer> list = new ArrayList(Arrays.asList(first, nums[left], nums[right]));
                    res.add(list);
                    left++;
                    right--;

                    while(left < right && nums[left] == nums[left - 1]) left++;
                    while(left < right && nums[right] == nums[right + 1]) right--;
                }else if(nums[left] + nums[right] < target){
                    left++;
                }else{
                    right--;
                }
            }
        }

        return res;
    }

}
```

### 10、二叉树的最近公共祖先

**时间复杂度O(N)：**其中N是二叉树的节点数。二叉树的所有节点有且只会被访问一次，因此时间复杂度为 O(N)。

**空间复杂度O(N)：**其中N是二叉树的节点数。递归调用的栈深度取决于二叉树的高度，二叉树最坏情况下为一条链，此时高度为 N，因此空间复杂度为 O(N)。

```java
class Solution {
    public TreeNode lowestCommonAncestor(TreeNode root, TreeNode p, TreeNode q) {
        if(root==null || root == p || root == q){
            return root;
        }

        TreeNode left = lowestCommonAncestor(root.left, p, q);
        TreeNode right = lowestCommonAncestor(root.right, p, q);

        //左右子树的返回值不为null，由于值唯一左右子树的返回值就是p和q，此时root为LCA
        if(left != null && right != null){
            return root;
        }
        //左右子树均返回空，p和q均不在树中，返回null
        if(left == null && right == null){
            return null;
        }
        //一方不为空说明p和q存在于左或右子树中。
        return left == null ? right : left;
    }
}
```

```java
class Solution {
    public int search(int[] nums, int target) {
        int left = 0;
        int right = nums.length - 1;

        while(left <= right){
            int mid = left + ((right - left) >> 1);

            if(target == nums[mid]){
                return mid;
            }
            // 左边有序
            if(nums[left] <= nums[mid]){
                //等号容易遗漏
                if(nums[left] <= target && nums[mid] >= target){
                    right = mid - 1;
                }else{
                    left = mid + 1;
                }
            }else{
                //等号容易遗漏
                if(nums[mid] <= target && target <= nums[right]){
                    left = mid + 1;
                }else{
                    right = mid - 1;
                }
            }
        }
        return -1;
    }
}
```

```java
        }
```

```java
    public int searchLeft(int[] nums, int target){
        int left = 0;
        int right = nums.length - 1;

        while(left <= right){
            int mid = left + ((right - left) >> 1);
            if(nums[mid] < target){
                left = mid + 1;
            }else if(nums[mid] >= target){
                right = mid - 1;
            }
        }

        if(left >= nums.length || nums[left]!=target){
                return -1;
        }

        return left;
    }

    public int searchRight(int[] nums, int target){
        int left = 0;
        int right = nums.length - 1;

        while(left <= right){
            int mid = left + ((right - left) >> 1);
            if(nums[mid] <= target){
                left = mid + 1;
            }else if(nums[mid] > target){
                right = mid - 1;
            }
        }

        if(right < 0 || nums[right] != target){
            return -1;
        }

        return right;
```

```java
class Solution {
    public List<Integer> rightSideView(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        if(root==null){
            return res;
        }

        Queue<TreeNode> queue = new LinkedList<>();
        queue.offer(root);

            int size = queue.size();

                TreeNode temp = queue.poll();
```

```java
class Solution {
    public int maximalSquare(char[][] matrix) {
        if(matrix == null || matrix.length == 0 || matrix[0].length == 0){
            return 0;
        }
        int m = matrix.length;
        int n = matrix[0].length;

        int[][] dp = new int[m + 1][n + 1];
        int max = 0;
        for(int i = 1;i < m + 1; i++){
            for(int j = 1; j < n + 1;j++){
                if(matrix[i - 1][j - 1] == '1'){
                    dp[i][j] = 1 + Math.min( dp[i - 1][j - 1], (Math.min(dp[i - 1][j], dp[i][j-1])));

                }
            }
        }
```

### 15、最长有效括号

### 16、基本计算器2

思路：默认第一个数为正数 sign = '+'，数d = 0。每次读到 '0'=>'9'：计算d，遇到 +，-就直接将d，-d压入栈中，遇到 * / 先出栈，再运算，再入栈。最后将栈内的所有数加起来就是结果了。  

```java
class Solution {
    public int calculate(String s) {
        int res = 0;
        int d = 0;
        int sign = '+'; //默认第一个为正数
        Stack<Integer> stack = new Stack<>();
        for(int i = 0;i < s.length();i++){
            Character c = s.charAt(i);
            if(c >= '0' && c <= '9'){
                d = d * 10 + c - '0';
            }

            if(c == '+' || c == '-' || c == '*' || c == '/' || i == s.length()-1){
                if(sign == '+'){
                    stack.push(d);
                }else if(sign == '-'){
                    stack.push(-d);
                }else if(sign == '*' || sign == '/'){
                    int temp = stack.pop();
                    temp = sign == '*' ? temp * d : temp / d;
                    stack.push(temp);
                }

                sign = c;
                d = 0;
            }
        }

        while(!stack.empty()){
            res += stack.pop();
        }

        return res;
    }
}
```

```java
        }else{
            l2.next = merge(l1, l2.next);
            return l2;
        }
    }


}
```

```java
class Solution {
    List<List<Integer>> res = new ArrayList<>(); 
    public List<List<Integer>> combinationSum(int[] candidates, int target) {
        if(candidates == null || candidates.length == 0){
            return res;
        }
        Arrays.sort(candidates);
        backward(new ArrayList(), candidates, target);
        return res;
    }

    public void backward(List<Integer> list, int[] candidates, int target){
        if(target <= 0){
            if(target == 0){
                res.add(new ArrayList(list));
            }
            return;
        }

        for(int i = 0;i < candidates.length;i++){
            if(list.size() > 0 && list.get(list.size() - 1) > candidates[i]){
                continue;
            }

            list.add(candidates[i]);
            backward(list, candidates, target - candidates[i]);
            list.remove(list.size()-1);
        }
    }
    
}
```

### 22、[ 单词搜索](https://leetcode-cn.com/problems/word-search/)

`dfs+回溯`

```java
class Solution {
    public boolean exist(char[][] board, String word) {
        if(board == null || board.length == 0 || board[0].length == 0){
            return false;
        }

        for(int i = 0 ;i < board.length;i++){
            for(int j = 0;j < board[0].length;j++){
                if(board[i][j] == word.charAt(0)){
                    if(dfs(board, word, i, j, 0)){
                        return true;
                    }
                }
            }
        }

        return false;

    }

    public boolean dfs(char[][] board, String word, int x, int y, int z){
        if(z == word.length()){
            return true;
        }
        if(x < 0 || x >= board.length || y < 0 || y >= board[0].length){
            return false;
        }

        if(word.charAt(z) != board[x][y]){
            return false;
        }
        char temp = board[x][y];
        board[x][y] = '0';
        
        boolean left = dfs(board, word, x - 1, y, z + 1);
        boolean right = dfs(board, word, x + 1, y, z + 1);
        boolean up = dfs(board, word, x, y - 1, z + 1);
        boolean down = dfs(board, word, x, y + 1, z + 1);
        board[x][y] = temp;
        return left || right || up || down;
    }
}
```

### 23、[旋转链表](https://leetcode-cn.com/problems/rotate-list/)

思路：串成循环链表，然后找到k位置的前一个节点，断开就好了。

```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode rotateRight(ListNode head, int k) {
        if(head == null || head.next == null){
            return head;
        }

        int len = 0;
        ListNode p = head;

        ListNode pre = new ListNode(0);
        pre.next = head;
        while(p != null){
            pre = pre.next;
            p = p.next;
            len++;
        }

        pre.next = head;
        for(int i = 0;i < len - (k % len);i++){
            pre = pre.next;
        }

        ListNode res = pre.next;
        pre.next = null;
        return res;
    }
}
```

```java
class Solution {
    List<List<Integer>> res = new ArrayList<>();
    public List<List<Integer>> subsets(int[] nums) {
        if(nums == null || nums.length == 0){
            return res;
        }

        helper(new ArrayList(), nums, 0);
        return res;

    public void helper(List<Integer> list, int[] nums, int start){
        res.add(new ArrayList(list));

            helper(list, nums, i + 1);
            list.remove(list.size()-1);
        }
    }
}
```

```java
class Solution {
    public int uniquePaths(int m, int n) {
        int[][] dp = new int[m][n];
        dp[0][0] = 1;

            dp[i][0] = 1;
```
